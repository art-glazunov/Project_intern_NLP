def preprocessing(text):
  #Prepares data using my module
                 
  text_res = my_preprocessing.text_early_preproc(text,del_html = True,del_punct_sp_chars=True,
                 del_underscore=False, del_digits=False) #delete punctuation and artifacts

  text_res = my_preprocessing.lemmatize_lower_case2(text_res)
 
  text_res = my_preprocessing.delete_stop_words(text_res)

  return text_res
